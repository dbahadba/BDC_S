CREATE  OR REPLACE  TRIGGER TR_GER_ESTRUTURA_LOJAPRODUTO1
 AFTER INSERT
 ON LOJA_PRODUTO_TT
 
 for each row
 
DECLARE
iGRUPO   GERENC.GER_GRUPO.GRUPO%TYPE;
iQTD     NUMBER;
dDT_SIG  DATE;


BEGIN

  SELECT COUNT(1)
  INTO iQTD
  FROM GERENC.GER_ESTRUTURA_PRODUTO
  WHERE EMP = :NEW.EMP AND LOJA = :NEW.LOJA AND
        FILIAL = :NEW.FILIAL AND PRODUTO = :NEW.PRODUTO;

  IF iQTD = 0 THEN

     SELECT GRUPO INTO iGRUPO FROM GERENC.GER_GRUPOS WHERE PRODUTO = :NEW.PRODUTO;

     iQTD := 0;

     SELECT COUNT(1)
     INTO iQTD
     FROM GERENC.GER_ESTRUTURA_PRODUTO
     WHERE EMP = :NEW.EMP AND LOJA = :NEW.LOJA AND
           FILIAL = :NEW.FILIAL AND GRUPO = iGRUPO;

     IF iQTD = 0 THEN

	       INSERT INTO GERENC.GER_ESTRUTURA_PRODUTO
		        (EMP,LOJA,FILIAL,PRODUTO,GRUPO,DT_TOOLS)
	       VALUES
		       (
		       :NEW.emp,
		       :NEW.loja,
		       :NEW.filial,
		       :NEW.produto,
		       iGRUPO,
           SYSDATE
		       );

     ELSE
         SELECT MIN(DT_SIG)
         INTO dDT_SIG
         FROM GERENC.GER_ESTRUTURA_PRODUTO

CREATE  OR REPLACE  TRIGGER TR_GER_ESTRUTURA_LOJAPRODUTO2
 AFTER DELETE
 ON LOJA_PRODUTO_TT
 
 for each row
 
DECLARE
iGRUPO     GERENC.GER_GRUPO.GRUPO%TYPE;
iQTD       NUMBER;
dDT_TOOLS  DATE;

BEGIN

  SELECT GRUPO INTO iGRUPO FROM GERENC.GER_GRUPOS WHERE PRODUTO = :OLD.PRODUTO;

  SELECT MIN(DT_TOOLS)
  INTO dDT_TOOLS
  FROM GERENC.GER_ESTRUTURA_PRODUTO
  WHERE EMP = :NEW.EMP AND LOJA = :NEW.LOJA AND
        FILIAL = :NEW.FILIAL AND PRODUTO = :NEW.PRODUTO;

  INSERT INTO GERENC.GER_ESTRUTURA_INATIVA
	       (EMP,LOJA,FILIAL,PRODUTO,DT_TOOLS,DT_INATIVO)
	VALUES
		     (
		     :OLD.emp,
		     :OLD.loja,
		     :OLD.filial,
		     :OLD.produto,
         dDT_TOOLS,
         SYSDATE
		     );

 DELETE FROM GERENC.GER_ESTRUTURA_PRODUTO
 WHERE EMP = :OLD.EMP AND LOJA = :OLD.LOJA AND
       FILIAL = :OLD.FILIAL AND PRODUTO = :OLD.PRODUTO;

  SELECT COUNT(1)
  INTO iQTD
  FROM GERENC.GER_ESTRUTURA_PRODUTO
  WHERE EMP = :OLD.EMP AND LOJA = :OLD.LOJA AND
        FILIAL = :OLD.FILIAL AND GRUPO = iGRUPO;

  IF iQTD  = 0 THEN
     UPDATE GERENC.GER_ESTRUTURA
     SET FLG_ATIVO = 'N',
         FLG_ALTERACAO = 'S'
     WHERE EMP = :OLD.EMP AND LOJA = :OLD.LOJA AND
           FILIAL = :OLD.FILIAL AND GRUPO = iGRUPO;

  END IF;

end TR_GER_ESTRUTURA_LOJAPRODUTO2;




/



ALTER TRIGGER TR_GER_ESTRUTURA_LOJAPRODUTO2
	DISABLE;
